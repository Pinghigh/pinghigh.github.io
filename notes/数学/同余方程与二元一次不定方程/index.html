i64 ex_gcd(i64 u, i64 v, i64& x, i64& y) {
    if (!v) {
        x = 1, y = 0;
        return u;
    }

    i64 g = ex_gcd(v, u % v, x, y);
    i64 temp = x;
    x = y;
    y = temp - u / v * y;
    return g;
}
