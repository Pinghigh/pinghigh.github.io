i32 tmp;

void exgcd(i32 a, i32 b, i32& x, i64& y) {
    if (!b) {
        x = 1, y = 0;
        return;
    }
    exgcd(b, a % b, x, y);
    tmp = x;
    x = y;
    y = tmp - (a / b) * y;
}

#define N 5000001

i32 a[N], s[N], f[N];
i64 n, p, k, ans, ktmp, y;

int main() {
    read(n, p, k);
    s[0] = 1;
    for (i32 i = 1; i <= n; ++i)
        read(a[i]), s[i] = s[i - 1] % p * a[i] % p;
    exgcd(s[n], p, f[n], y);
    // write(p, '\n');
    for (i32 i = n - 1; i; --i) {
        f[i] = f[i + 1] % p * a[i + 1] % p;
    }
    f[1] %= p;
    ans = k * f[1] % p;
    ktmp = k % p;
    for (i32 i = 2; i <= n; ++i) {
        f[i] = s[i - 1] % p * f[i] % p;
        ktmp = (ktmp * k) % p;
        ans = ((ans % p + ktmp * f[i]) % p + p) % p;
    }
    write(ans, '\n');
    
    return 0;
}
